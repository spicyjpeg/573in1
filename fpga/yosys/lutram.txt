ram distributed \_RAM16X1S {
	abits 4;
	width 1;
	cost  1; # Uses a single F or G LUT
	init  no_undef;
	prune_rom;

	port arsw "0" { clock posedge; }
}

ram distributed \_RAM16X1D {
	abits 4;
	width 1;
	cost  2; # Uses both F and G LUTs
	init  no_undef;
	prune_rom;

	port arsw "0" { clock posedge; }
	port ar   "1" { }
}

ram distributed \_RAM32X1S {
	abits 5;
	width 1;
	cost  2; # Uses both F and G LUTs
	init  no_undef;
	prune_rom;

	port arsw "0" { clock posedge; }
}
